### 🧩 `script.js`

```javascript
// Get city from query string, default to Albany
const params = new URLSearchParams(window.location.search);
const cityKey = params.get("city")?.toLowerCase() || "albany";

// Get the itinerary for the city
const itinerary = itineraries[cityKey];

if (itinerary) {
  document.getElementById("title").textContent = itinerary.title;
  document.getElementById("location").textContent = "📍 " + itinerary.location;

  document.getElementById("details").innerHTML = `
    <p><strong>Theme:</strong> ${itinerary.theme}</p>
    <p><strong>Style:</strong> ${itinerary.style}</p>
    <p><strong>Trip Length:</strong> ${itinerary.length}</p>
    <p><strong>Best For:</strong> ${itinerary.bestFor}</p>
  `;

  document.getElementById("view-button").href = itinerary.link;
} else {
  document.getElementById("title").textContent = "City Not Found";
  document.getElementById("location").textContent = "Try a different city in the URL.";
  document.querySelector(".details").innerHTML = "";
}
```
